# 通用表格

near-admin提供了一个通用的表格组件，内置行距调整、列选择、全屏展示的功能，方便开发者提供高质量的搜索结果列表

# 使用方法

// 同步
import NCommonTable from '@corecp/NCommonTable.vue'
// 异步
const NCommonTable = () => import('@corecp/NCommonTable.vue')

# props

name value 说明
tableObj type: ComTable 需要传入表格对象，类型是coretype里面的ComTable

# typescript type

interface ComTableBtn {
    name: string,
    method: string,
    icon?: string,
    type?: string
}
interface ComTable {
    btnList?: ComTableBtn[],
    columns: any[],
    data: any[],
    title: string,
    pagingInfo: PagingInfo
}
interface PagingInfo {
    currentPage: number,
    pageSize: number,
    totalRows: number
}

# ComTableBtn

name value
btnList 按钮列表
columns 列
data 数据
title 鼠标移上去的提示
pagingInfo 分页参数

# ComTableBtn

name value
btnList 按钮列表
columns 列
data 数据
title 鼠标移上去的提示
pagingInfo 分页参数

# PagingInfo

name value
currentPage 当前页
pageSize 每页几条记录
totalRows 总过几条记录

# slot

name 说明
n-com-table                     用于插入ant-design的表格

# slot-scope

name value 说明
sizeClass n-com-table-small：小；n-com-table-normal：中；n-com-table-large：大 控制表格行距和字体大小，组件内置的样式，设置在a-table组件的class
selectColumns Array 选中的列数，设置在a-table的columns属性
rowClass stripe-light：亮；stripe-dark：暗 用于斑马纹表格，对应设置在a-table的rowClassName属性
innerHeight Number 表格在页面渲染完成之后，能够返回实际表格的高度，可以用于动态高度（如flex）的表格设置scroll里面的y值
resizeTitle Component 将返回的组件设置到表格ATable组件的components属性

# event

name 说明 参数
btnevent 点击按钮事件 method：方法名，可以使用this[method]()调用对应的方法

# 例子（代码摘自日志页Logline.vue）

<n-common-table :table-obj="loglineTable" @btnevent="btnEvent">
    <a-table slot-scope="{sizeClass, selectColumns, rowClass, innerHeight, resizeTitle}"
             :class="sizeClass" row-key="idx" slot="n-com-table"
             :columns="selectColumns"
             :data-source="loglineTable.data"
             :scroll="{ x: '100%', y: innerHeight }"
             :pagination="loglineTable.pagingInfo"
             :components="resizeTitle"
             :row-class-name="rowClass"
             @change="pageChange"
    >
        <a-tooltip placement="topLeft" slot="title-1">
            <template slot="title">
                {{ $t(dict.localeObj.logline.tableColumns.index) }}
            </template>
            <span>{{ $t(dict.localeObj.logline.tableColumns.index) }}</span>
        </a-tooltip>
        <a-tooltip placement="topLeft" slot="title-2">
            <template slot="title">
                {{ $t(dict.localeObj.logline.tableColumns.desc) }}
            </template>
            <span>{{ $t(dict.localeObj.logline.tableColumns.desc) }}</span>
        </a-tooltip>
        <a-tooltip placement="topLeft" slot="title-3">
            <template slot="title">
                {{ $t(dict.localeObj.logline.tableColumns.type) }}
            </template>
            <span>{{ $t(dict.localeObj.logline.tableColumns.type) }}</span>
        </a-tooltip>
        <a-tooltip placement="topLeft" slot="title-4">
            <template slot="title">
                {{ $t(dict.localeObj.logline.tableColumns.time) }}
            </template>
            <span>{{ $t(dict.localeObj.logline.tableColumns.time) }}</span>
        </a-tooltip>
        <a-tooltip placement="topLeft" slot="title-5">
            <template slot="title">
                {{ $t(dict.localeObj.logline.tableColumns.op) }}
            </template>
            <span>{{ $t(dict.localeObj.logline.tableColumns.op) }}</span>
        </a-tooltip>
        <span slot="time" slot-scope="time">
            {{ timestampToDateString(time, '-', false, false, false) }}
        </span>
        <span slot="operate" slot-scope="record">
            <a-button class="n-ant-btn-link" type="link" @click="showLogDetail(record)">{{ $t(dict.localeObj.normalBtn.detail) }}</a-button>
        </span>
    </a-table>
</n-common-table>
