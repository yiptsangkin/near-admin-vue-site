# 多列下拉

near-admin多列下拉组件是基于a-select重新封装的一个组件，在支持select组件原本的功能基础上，增加了多列展示、分页的功能，类似datagrid的功能。方便 有类似使用场景的开发者进行使用

# 使用方法

// 同步
import NMultiSelect from '@corecp/NMultiSelect.vue'
// 异步
const NMultiSelect = () => import('@corecp/NMultiSelect.vue')

# props

name value 说明
optionColumns type: MultiSelectColumn 需要在下拉框里面展示的列
showTitleField true：展示 / false：不展示 是否在列标题里面展示列对应的字段
showSortIndex true：展示 / false：不展示 是否展示序号
optionFormatter 选项前置处理函数 用于处理选项列表
pagingInfo type: PagingInfo 分页信息
value(v-model) Any 选中的value值
valueProps string 默认'value'
remote true：远程请求返回数据 / false：静态数据 为静态数据的时候，组件内部会根据pagingInfo在前端分页，并计算总记录数，远程请求返回数据则不会

# typescript type

interface MultiSelectColumn {
    title: string,
    field: string,
    width?: string | number
}
interface PagingInfo {
    currentPage: number,
    pageSize: number,
    totalRows: number
}

# MultiSelectColumn

name value
title 列名
field 列字段
width 列宽

# PagingInfo

name value
currentPage 当前页
pageSize 每页几条记录
totalRows 总过几条记录

# event

name 说明 参数
changepage 分页切换事件 page：页数，返回当前页

# 例子

<template>
    <div class="customer-entry">
        <a-row>
            <a-col :span="10">
                <template v-if="cacheEntry.length > 0">
                    <n-common-title :title="dict.localeObj.setting.customEntry" no-top></n-common-title>
                    <div class="customer-entry-item-wrp">
                        <span @click="toPage(item)" class="customer-entry-item" v-for="(item, index) in cacheEntry" :key="index" :span="6">
                            <a-tooltip placement="top" slot="title-1">
                                <template slot="title">
                                    {{ $t(item.title) }}
                                </template>
                                {{ $t(item.title) }}
                            </a-tooltip>
                        </span>
                    </div>
                </template>
            </a-col>
        </a-row>
        <a-row>
            <n-multi-select
                    v-model="abe"
                    :paging-info="pagingInfo"
                    :options-list="optionsList"
                    :option-columns="optionColumns"
                    show-search
                    @changepage="changePage"
            ></n-multi-select>
        </a-row>
    </div>
</template>

<script lang="ts">
    import Vue from 'vue'
    import CoreBase from '@corets/base'
    import Base from '@custom/base'
    import {mapGetters} from 'vuex'
    import NCommonTitle from '@corecp/NCommonTitle.vue'
    import NMultiSelect from '@corecp/NMultiSelect.vue'
    import {PagingInfo} from '@corets/type';

    export default Vue.extend({
        name: 'HomePage',
        mixins: [CoreBase, Base],
        components: {
            NCommonTitle,
            NMultiSelect
        },
        computed: {
            ...mapGetters([
                'cacheEntry',
                'userInfo'
            ])
        },
        methods: {
            toPage (page: any) {
                const self = this as any
                self.$newpage({
                    title: page.title,
                    component: page.path
                })
            },
            changePage (page: any) {
                console.log(page)
            }
        },
        data () {
            return {
                optionsList: [
                    {
                        value: 'nihao1',
                        label: 'nihao2'
                    }
                ],
                test: '',
                optionColumns: [
                    {
                        title: '你好',
                        field: 'value',
                        width: 240
                    },
                    {
                        title: '你好',
                        field: 'label',
                        width: 240
                    }
                ],
                abe: 'nihao1',
                pagingInfo: {
                    currentPage: 1,
                    pageSize: 10
                } as PagingInfo
            }
        }
    })
</script>
