# 不想使用typescript

如果你不想使用typescript，又想使用near-admin可以吗？答案当然是可以的，你只需要在把sfc里面script标签上的lang="ts"去掉即可，同时页面内部不 需要再使用Vue.extend去包裹实例参数。

例如我们把homepage的内容改成js，也是可以正常运行的。但是这里作者还是推荐使用ts进行编写，哪怕用成了anyscript，因为类型的代码提示的确能够进行一定 代码规范和限制。

<template>
    <div class="customer-entry">
        <a-row>
            <a-col :span="10">
                <template v-if="cacheEntry.length > 0">
                    <n-common-title :title="dict.localeObj.setting.customEntry" no-top></n-common-title>
                    <div class="customer-entry-item-wrp">
                        <span @click="toPage(item)" class="customer-entry-item" v-for="(item, index) in cacheEntry" :key="index" :span="6">
                            <a-tooltip placement="top" slot="title-1">
                                <template slot="title">
                                    {{ $t(item.title) }}
                                </template>
                                {{ $t(item.title) }}
                            </a-tooltip>
                        </span>
                    </div>
                </template>
            </a-col>
        </a-row>
    </div>
</template>

<script>
    import CoreBase from '@corets/base'
    import Base from '@custom/base'
    import {mapGetters} from 'vuex'
    import NCommonTitle from '@corecp/NCommonTitle.vue'
    import NMultiSelect from '@corecp/NMultiSelect.vue'

    export default {
        name: 'HomePage',
        mixins: [CoreBase, Base],
        components: {
            NCommonTitle,
            NMultiSelect
        },
        computed: {
            ...mapGetters([
                'cacheEntry',
                'userInfo'
            ])
        },
        methods: {
            toPage (page) {
                const self = this
                self.$newpage({
                    title: page.title,
                    component: page.path
                })
            }
        },
        data () {
            return {
            }
        }
    }
</script>

<style lang="scss" scoped>
    @import '~@scss/custom/manage/homepage.scss'
</style>
